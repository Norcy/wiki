void quickSort(int s[], int l, int r)
{
    if (l >= r)
    {
        return;
    }
    
    int i = l, j = r, x = s[l];
    
    while (i < j)
    {
        while (s[j] >= x && i < j)
        {
            j--;
        }

        if (i < j)
        {
            s[i++] = s[j];
        }

        while (s[i] < x && i < j)
        {
            i++;
        }

        if (i < j)
        {
            s[j--] = s[i];
        }
    }

    s[i] = x;
    quickSort(s, l, i-1);
    quickSort(s, i+1, r);
}
